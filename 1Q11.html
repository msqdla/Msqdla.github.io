        // 显示详情
        function showDetail(npc) {
            document.getElementById('detailName').textContent = npc.name;
            document.getElementById('detailLevel').textContent = `${npc.class} ${npc.level}`;
            document.getElementById('detailFaction').textContent = npc.faction;
            document.getElementById('detailBio').textContent = npc.bio;
            document.getElementById('detailBackground').textContent = npc.background;
           
            // 生成详细属性
            const statsContainer = document.getElementById('detailStats');
            statsContainer.innerHTML = '';
           
            for (const [key, value] of Object.entries(npc.stats)) {
                const statItem = document.createElement('div');
                statItem.className = 'detail-stat-item';
                statItem.innerHTML = `
                    <div class="detail-stat-label">${key}</div>
                    <div class="detail-stat-value">${value}</div>
                `;
                statsContainer.appendChild(statItem);
            }
           
            document.getElementById('detailModal').style.display = 'block';
        }

        // 关闭模态框
